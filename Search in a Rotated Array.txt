        // l: The starting index
        // h: The ending index, you have to search the key in this range
        
        //complete the function here
        
        //int c = 0;
        for(int i = l ; i<=h ; i++){
             if(l==0 && h==0){
                 if(A[0]==key){
                 return 0;
                 }
                 else{
                     return -1;
                 }
            }
            if(A[i]== key){
                return i;
            }
        }
        return -1;

    }
};
